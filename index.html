2. Control Structures

2.1 If-Else


#include <iostream>

int main() {
    int number;
    std::cout << "Enter a number: ";
    std::cin >> number;

    if (number > 0) {
        std::cout << "Positive number" << std::endl;
    } else if (number < 0) {
        std::cout << "Negative number" << std::endl;
    } else {
        std::cout << "Zero" << std::endl;
    }

    return 0;
}

2.2 Loops

For Loop


#include <iostream>

int main() {
    for (int i = 1; i <= 5; i++) {
        std::cout << "Number: " << i << std::endl;
    }
    return 0;
}

While Loop


#include <iostream>

int main() {
    int i = 1;
    while (i <= 5) {
        std::cout << "Number: " << i << std::endl;
        i++;
    }
    return 0;
}

2.3 Switch Case


#include <iostream>

int main() {
    char grade;
    std::cout << "Enter grade (A/B/C): ";
    std::cin >> grade;

    switch (grade) {
        case 'A': std::cout << "Excellent!" << std::endl; break;
        case 'B': std::cout << "Good!" << std::endl; break;
        case 'C': std::cout << "Fair!" << std::endl; break;
        default: std::cout << "Invalid grade!" << std::endl;
    }
    return 0;
}

3. Functions

3.1 Function Definition


#include <iostream>

int add(int a, int b) {
    return a + b;
}

int main() {
    int sum = add(5, 10);
    std::cout << "Sum: " << sum << std::endl;
    return 0;
}

3.2 Function Overloading


#include <iostream>

int add(int a, int b) {
    return a + b;
}

double add(double a, double b) {
    return a + b;
}

int main() {
    std::cout << "Int sum: " << add(5, 10) << std::endl;
    std::cout << "Double sum: " << add(5.5, 10.5) << std::endl;
    return 0;
}

4. Object-Oriented Programming (OOP)

4.1 Classes and Objects


#include <iostream>

class Car {
public:
    std::string model;
    int year;

    void displayInfo() {
        std::cout << "Model: " << model << ", Year: " << year << std::endl;
    }
};

int main() {
    Car car1;
    car1.model = "Toyota";
    car1.year = 2020;
    car1.displayInfo();
    return 0;
}

4.2 Inheritance


#include <iostream>

class Animal {
public:
    void speak() {
        std::cout << "Animal speaks" << std::endl;
    }
};

class Dog : public Animal {
public:
    void speak() {
        std::cout << "Dog barks" << std::endl;
    }
};

int main() {
    Dog dog;
    dog.speak();
    return 0;
}

4.3 Polymorphism


#include <iostream>

class Shape {
public:
    virtual void draw() {
        std::cout << "Drawing a shape" << std::endl;
    }
};

class Circle : public Shape {
public:
    void draw() override {
        std::cout << "Drawing a circle" << std::endl;
    }
};

int main() {
    Shape* shape = new Circle();
    shape->draw();
    return 0;
}

5. Advanced Concepts

5.1 Smart Pointers


#include <iostream>
#include <memory>

class Test {
public:
    void sayHello() {
        std::cout << "Hello, Smart Pointer!" << std::endl;
    }
};

int main() {
    std::unique_ptr ptr = std::make_unique();
    ptr->sayHello();
    return 0;
}

5.2 Lambda Expressions


#include <iostream>

int main() {
    auto add = [](int a, int b) { return a + b; };
    std::cout << "Sum: " << add(5, 7) << std::endl;
    return 0;
}

6. STL (Standard Template Library)

6.1 Vectors


#include <iostream>
#include <vector>

int main() {
    std::vector<int> vec = {1, 2, 3, 4, 5};
    for (int num : vec) {
        std::cout << num << " ";
    }
    return 0;
}

6.2 Maps


#include <iostream>
#include <map>

int main() {
    std::map<std::string, int> map;
    map["apple"] = 5;
    map["banana"] = 3;

    for (auto &pair : map) {
        std::cout << pair.first << ": " << pair.second << std::endl;
    }

    return 0;
}
